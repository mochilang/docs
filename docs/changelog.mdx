import releases from '@site/src/data/releases.json';

export const latest = releases.slice(0, 12);
export const trains = releases.reduce((acc, release) => {
  const train = `v${release.major}.${release.minor}`;
  if (!acc[train]) acc[train] = [];
  acc[train].push(release);
  return acc;
}, {});

The website changelog is generated from the Markdown release notes in the
repository's [`releases/`](https://github.com/mochilang/mochi/tree/main/releases)
directory. Each release has its own page, so this index can stay readable while
the full patch history remains available on the site.

## Upgrade path

Start with the newest release page, then scan backwards through the release
train that you are currently using. Patch releases in the same train are
usually cumulative: compiler and runtime fixes, example corpus growth, docs
updates, and release engineering improvements.

For a raw chronological file, use
[CHANGELOG.md](https://github.com/mochilang/mochi/blob/main/CHANGELOG.md) on
GitHub. For source release notes, use the
[`releases/`](https://github.com/mochilang/mochi/tree/main/releases) directory.
